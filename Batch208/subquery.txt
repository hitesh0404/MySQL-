create view sales as 
 select
	customernumber,
	customerName,
	sum(total) from ( 
		         select 
			c.customerNumber,
			customerName,
			o.orderNumber,
			sum(quantityOrdered*priceEach) as total 
		         from
  			customers c 
		         natural join 
			orders o  
		         natural join 
			orderdetails 
    		         group by 
			o.ordernumber ) as sales  

	group by 
		customerNumber 
	order by sum(total) desc limit 10;
