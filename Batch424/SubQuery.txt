select 
	child.*,
	parent.employeenumber,
	parent.firstname,
	parent.lastname,
	child.reportsTo
from 
	employees as child 
join 
	employees as parent
on 
	parent.employeenumber = child.reportsTo;

(select  
	c.customerNumber

	customers c 
join 
	orders using(customernumber) 
join 
	sum_of_value_of_orders using(ordernumber) 

group by customernumber

sum(total_value)>500000);
